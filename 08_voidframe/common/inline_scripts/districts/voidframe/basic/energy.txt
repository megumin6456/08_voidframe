district_giga_voidframe_generator_$tier$ = {
	base_buildtime = $time$
	icon = district_giga_voidframe_generator_$tier$
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		uses_district_set = giga_frameworld
		not = { has_planet_flag = frameworld_penrose }
		is_voidframe_exclusive_$tier$ = yes
	}
	
	potential = {
		uses_district_set = giga_frameworld
		not = { has_planet_flag = frameworld_penrose }
		if = {
			limit = { is_voidframe_outdated_$tier$ = yes }
			$pass_condition$
			NOR = { $fail_condition$ }
		} else = {
			is_voidframe_$tier$ = yes
			OR = {
				$pass_condition$
				$fail_condition$
			}
		}
	}

	prerequisites = {
		$reqtech$
	}

	show_tech_unlock_if = {
		giga_has_frameworld_origin = yes
	}
	
	conversion_ratio = 1
	convert_to = {
		district_giga_frameworld_generator
		district_giga_frameworld_penrose

		district_giga_voidframe_generator_dm
		district_giga_voidframe_generator_de
		district_giga_voidframe_generator_pe
		district_giga_voidframe_generator_se

		district_giga_voidframe_penrose_dm
		district_giga_voidframe_penrose_de
		district_giga_voidframe_penrose_pe
		district_giga_voidframe_penrose_se
	}
	
	resources = {
		category = planet_districts
		cost = {
			$cost$
		}
		upkeep = {
			$upkeep$
		}
	}
	
	planet_modifier = {
		planet_housing_add = $house$
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}
	
	# ACOT
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
			}
			OR = {
				has_voidframe_mods_aot = no
				always = $ignore_aot$
			}
		}
		modifier = {
			$acot_normjob$
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = yes
			}
			OR = {
				has_voidframe_mods_aot = no
				always = $ignore_aot$
			}
		}
		modifier = {
			$acot_gestjob$
		}
	}

	# AOT
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
			}
			has_voidframe_mods_aot = yes
		}
		modifier = {
			$aot_normjob$
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = yes
			}
			has_voidframe_mods_aot = yes
		}
		modifier = {
			$aot_gestjob$
		}
	}
}
