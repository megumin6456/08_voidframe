decision_giga_voidframe_$type$_downgrade_$tier$ = {
	# this means downgrade to tier
	# THE NAME IS THE TARGET
	icon = decision_giga_voidframe_$type$_$tier$
	owned_planets_only = yes

	# SCRIPT INPUTS
	# type = district type
	# tier = tier that we downgrade to
	# tier_up = level above $tier$ (level we are downgrading FROM)
	# cost = cost of upgrading

	potential = {
		# standard frame checks
		exists = owner
		owner = {
			giga_has_frameworld_origin = yes
		}
		uses_district_set = giga_frameworld

		NOT = {
			# doesnt have the district flag that we downgrade to
			has_planet_flag = anz_voidframe_$type$_$tier$
		}
		
		# no need to check for frame lvl, we are downgrading

		# check if we have the flag to downgrade from
		has_planet_flag = anz_voidframe_$type$_$tier_up$
		
		has_planet_flag = anz_voidframe_show_downgrade
	}

	# no reqtech, we are downgrading

	resources = {
		category = megastructures
		cost = {
			$cost$
		}
	}

	effect = {
		custom_tooltip = decision_giga_voidframe_$type$_downgrade_$tier$_effect
		hidden_effect = {
			set_planet_flag = anz_voidframe_$type$_$tier$
			remove_planet_flag = anz_voidframe_$type$_$tier_up$
			validate_planet_buildings_and_districts = yes
		}
	}
	
	ai_weight = {
		weight = 10
	}
}
