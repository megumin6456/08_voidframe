@defensive_station_distance = 75

#GENERIC

# destroy old defensive stations

anz_voidframe_destroy_old_defensive_stations = {
    solar_system = {
        every_ship_in_system = {
            limit = {
                has_ship_flag = giga_frameworld_defensive_station
            }
            destroy_ship = this
        }
    }
}

#spawn a guardian at radial location
# $WHO$ = owner
# $WHERE$ = position
# $DISTANCE$ = how far away
# $ANGLE$ = at what radial angle

anz_voidframe_spawn_guardian_defensive_station = {
    $WHO$ = { save_event_target_as = voidframe_spawn_guardian_defensive_station_owner }
    $WHERE$ = { save_event_target_as = voidframe_spawn_guardian_defensive_station_location }

    create_fleet = {
        name = "voidframe_guardian_defensive_station_fleet"

        effect = {
            set_owner = event_target:voidframe_spawn_guardian_defensive_station_owner

            create_ship = {
                name = random
                random_existing_design = precursor_guardian
                upgradable = yes
                #THIS DESIGN IS UPGRADABLE, PERHAPS MAKE AN AUTO-UPGRADE ONE?
                #ALTHOUGH THIS LETS PEOPLE SET THE DIFFERENT SECTIONS, BUT REPAIR WONT BE ADJUSTED ACCORDINGLY
                #FIX THIS THROUGH TRIGGERED DECISION COST?

                effect = {
                    set_disable_at_health = 0
                    set_ship_flag = voidframe_guardian_defensive_station
                    set_ship_flag = voidframe_guardian_defensive_station_$ID$
                }
            }
            set_location = {
                target = event_target:voidframe_spawn_guardian_defensive_station_location
                distance = $DISTANCE$
                angle = $ANGLE$
                direction = out_system
            }

            set_fleet_flag = voidframe_guardian_defensive_station
            set_fleet_flag = voidframe_guardian_defensive_station_$ID$
        }
    }
}

# check if this destroyed station is a specific ID, and do death
# this = destroyed station fleet
# $ID$ = which id to check against
anz_voidframe_guardian_defensive_station_id_destroyed = {
    if = {
        limit = {
            has_ship_flag = voidframe_guardian_defensive_station_$ID$
        }

        # create numbered wreckage
        create_ambient_object = {
            type = giga_frameworld_defensive_station_wreckage
            use_3d_location = yes
            location = this
        }
        last_created_ambient_object = {
            set_ambient_object_flag = voidframe_guardian_defensive_station_$ID$
        }

        # set destroyed flag on frameworld
        solar_system = {
            random_system_planet = {
                limit = {
                    giga_is_frame_world = yes
                }
                set_planet_flag = voidframe_guardian_defensive_station_$ID$_destroyed
            }
        }

        destroy_ship = this
    }
}

# respawn a defensive station at wreckage of the same ID
# this = planet
# $ID$ = which ID to check
# $WHO$ = owner
anz_voidframe_respawn_guardian_defensive_station = {
    solar_system = {
        every_system_ambient_object = {
            limit = {
                has_ambient_object_flag = voidframe_guardian_defensive_station_$ID$
            }

            destroy_ambient_object = this
        }
    }

    anz_voidframe_spawn_guardian_defensive_station_$ID$ = { WHO = $WHO$ }

    remove_planet_flag = voidframe_guardian_defensive_station_$ID$_destroyed
}

# check if a station with a specific ID is missing, if it is set flag on PLANET
# this = system
# $PLANET$ = the frameworld to set the flag on
# $ID$ = which ID to check
anz_voidframe_check_guardian_defensive_station = {
    if = {
        limit = {
            not = {
                any_fleet_in_system = {
                    has_fleet_flag = voidframe_guardian_defensive_station_$ID$
                }
            }
            not = {
                $PLANET$ = {
                    has_planet_flag = voidframe_guardian_defensive_station_$ID$_destroyed
                }
            }
        }

        $PLANET$ = {
            set_planet_flag = voidframe_guardian_defensive_station_$ID$_destroyed
        }
    }
}
