# calculate number of bunker districts for making armies
maginot_bunker_count = {
    base = 1

    complex_trigger_modifier = {
        trigger = num_districts
        parameters = { type = district_maginot_world_bunkers }
        mode = add
    }

    complex_trigger_modifier = {
        trigger = num_districts
        parameters = { type = district_maginot_ringworld_bunkers }
        mode = add
        mult = 5
    }

    complex_trigger_modifier = {
        trigger = num_districts
        parameters = { type = district_giga_frameworld_fortress_bunker }
        mode = add
    }

    #### multiplies if the zone is there
    modifier = {
        mult = 2
        has_zone = { zone = zone_voidframe_zone_DM_fortress_maginot }
    }
    modifier = {
        mult = 2
        has_zone = { zone = zone_voidframe_zone_DE_fortress_maginot }
    }
    modifier = {
        mult = 3
        has_zone = { zone = zone_voidframe_zone_PE_fortress_maginot }
    }
    modifier = {
        mult = 3
        has_zone = { zone = zone_voidframe_zone_SE_fortress_maginot }
    }

}
